CREATE TABLE SALE_SPRING3(
	SALE_ID NUMBER(10),
	USER_ID VARCHAR2(50),
	UPDATE_TIME TIMESTAMP,
	PRIMARY KEY (SALE_ID,USER_ID),
	FOREIGN KEY (USER_ID) REFERENCES USER_ACCOUNT_SPRING3 (USER_ID)
);

CREATE TABLE SALE_LINE_SPRING3(
	SALE_ID NUMBER(10),
	SALE_LINE_ID NUMBER(3),
	USER_ID VARCHAR2(50),
	ITEM_ID NUMBER(5),
	QUANTITY NUMBER(1),
	UPDATE_TIME TIMESTAMP,
	PRIMARY KEY (SALE_ID,SALE_LINE_ID),
	FOREIGN KEY (SALE_ID,USER_ID) REFERENCES SALE_SPRING3 (SALE_ID,USER_ID),
	FOREIGN KEY (ITEM_ID) REFERENCES ITEM_SPRING3 (ITEM_ID)
);
